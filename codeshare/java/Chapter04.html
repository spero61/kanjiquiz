    package array;

    class TV{

     int year, inch;

     TV(String company,int year, int inch){

      this.company=company;

     }

     void show() {
      // (2) company에 해당하는 %s (String), year에 해당하는 %d(십진수 숫자)는 있는데 inch 에 해당하는 %d가 없음
      System.out.printf("%s에서 만든 %d년형 %인치 TV",company,year,inch);

     public class Chapter04_1{

      // (3) 메인 함수는  public static void main(String[] args) { }

      public void main(String args[]) {
       TV myTV = new TV("LG",2017,32);
       myTV.show();

// 실습문제 4장 3번 문제

    // 참 잘 했습니다! 문제없이 의도한 바대로 잘 실행되는군요
    // 내용을 확실히 익히기 위해 몇 가지 체크하며 넘어갑시다.

    package array;


    class Song {

      // 여기에 선언된 변수들을 주로 멤버변수(member variable), 또는 필드(field)라고 부릅니다.
      private String title;
      private String artist;
      private int year;
      private String country;

      // 아무 내용도 없는 생성자를 만들어 주었습니다. 그 이유는 바로 아래 (String title, String artist, int year, String country)를
      // 파라미터로 입력받는 생성자를 만들어주면 저기 맨 아래 부분 메인 함수안에 "(여기!!)" 부분을 실행할 수 없기 때문입니다. 그 이유는 무엇일까요?
      Song() {};

      Song(String title, String artist, int year, String country) {
        this.title = title;
        this.artist = artist;
        this.year = year;
        this.country = country;
      }

      void show() {
        System.out.printf("%d년 %s국적의 %s가 부른 %s", year, country, artist, title);
      }
    }


    public class Ch4_3 {
      public static void main(String args[]) {
        Song mySong = new Song(); // <---------------(여기!!)

        mySong = new Song("Dancing Queen", "ABBA", 1978, "스웨덴");
        mySong.show();

      }
    }
