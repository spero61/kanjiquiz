        package assignment;

import java.sql.*;
import java.awt.*;
import javax.swing.*;
import javax.swing.table.*;



public class JdbcDynamicSQLFinal { // ADD버튼, REMOVE버튼

	public static void main(String[] args) {
		StringBuffer sqlSelect = new StringBuffer();
		sqlSelect.append("SELECT DEPTNO, S_ID, S_NAME, KOR, ENG, MATH ");
		sqlSelect.append("FROM STUDENT");

		Connection conn = null;
		Statement stmt = null;
		ResultSet rs = null;

		try {
			conn = ConnUtil.getConnection(); // ****************
			stmt = conn.createStatement();

			rs = stmt.executeQuery(sqlSelect.toString());

			int rowCount = 0;

			while (rs.next()) {
				++rowCount;

			}


		// --------------------공통 모듈---------------------
		JFrame jf = new JFrame("학생 성적 관리 프로그램");
		jf.setPreferredSize(new Dimension(1000, 700));
		jf.setLocation(500, 400);
		Container contentPane = jf.getContentPane();
		// -------------------------------------------------


		// ----------------------그래프 드로잉 패널--------------------------
		DrawingPanel drawingPanel = new DrawingPanel();
		contentPane.add(drawingPanel, BorderLayout.CENTER);
		// -----------------------------------------------------------


		// String colName[] = {"DEPTNO", "S_ID", "S_NAME", "KOR", "ENG", "MATH"};
		String colName[] = { "학과", "학번(8자리)", "이름", "국어", "영어", "수학" };

		// (colName : 컬럼 이름이 있는 1차원 배열, 0 : 행의 수);
		DefaultTableModel model = new DefaultTableModel(colName, 0);

		JTable table = new JTable(model);

		contentPane.add(new JScrollPane(table), BorderLayout.CENTER);

		// -------------------SELECT 문 받아서 출력
		rs = stmt.executeQuery(sqlSelect.toString());
		while (rs.next()) {
			Object arr[] = new Object[6];
			int col1 = rs.getInt(1); // 또는 rs.getInt("DEPTNO");
			int col2 = rs.getInt("S_ID"); // 또는 rs.getInt(2);
			String col3 = rs.getString("S_NAME"); // 또는 rs.getString(3);
			int col4 = rs.getInt(4);
			int col5 = rs.getInt(5);
			int col6 = rs.getInt(6);

			arr[0] = col1;
			arr[1] = col2;
			arr[2] = col3;
			arr[3] = col4;
			arr[4] = col5;
			arr[5] = col6;
			model = (DefaultTableModel) table.getModel();
			model.addRow(arr);
		}

		JPanel jp = new JPanel();

		// 패널 레이아웃 설정
		// jp.setLayout(new GridLayout(3,4));

		JTextField txt1 = new JTextField(2); // 입력창 필드 크기 설정
		JTextField txt2 = new JTextField(7);
		JTextField txt3 = new JTextField(5);
		JTextField txt4 = new JTextField(3);
		JTextField txt5 = new JTextField(3);
		JTextField txt6 = new JTextField(3);

		JLabel label1 = new JLabel(colName[0]);
		JLabel label2 = new JLabel(colName[1]);
		JLabel label3 = new JLabel(colName[2]);
		JLabel label4 = new JLabel(colName[3]);
		JLabel label5 = new JLabel(colName[4]);
		JLabel label6 = new JLabel(colName[5]);

		JButton btn1 = new JButton("추가");
		JButton btn2 = new JButton("삭제");
		JButton btn3 = new JButton("출력");
		JButton btn4 = new JButton("그래프");

		// 위 내용을 패널에 추가
		jp.add(label1);
		jp.add(txt1);

		jp.add(label2);
		jp.add(txt2);

		jp.add(label3);
		jp.add(txt3);

		jp.add(label4);
		jp.add(txt4);

		jp.add(label5);
		jp.add(txt5);

		jp.add(label6);
		jp.add(txt6);

		jp.add(btn1);
		jp.add(btn2);
		jp.add(btn3);
		jp.add(btn4);

		contentPane.add(jp, BorderLayout.SOUTH);


		// ADD Listener 등록 / constructor arguments 확인할 것
		btn1.addActionListener(new AddButtonActionListener(table, txt1, txt2, txt3, txt4, txt5, txt6));

		// REMOVE Listener 등록 / constructor arguments 확인할 것
		btn2.addActionListener(new RemoveButtonActionListener(table));

		// Print Listener 등록
		btn3.addActionListener(new PrintButtonActionListener(table));


		btn4.addActionListener(new DrawActionListener(txt4, txt5, txt6, drawingPanel));

		// --------------------공통 모듈2--------------------
		jf.setDefaultCloseOperation(jf.EXIT_ON_CLOSE);
		jf.pack();
		jf.setVisible(true);
		// -------------------------------------------------

	}catch (SQLException se) {
			se.printStackTrace();
		} finally {
			try {
				if (rs != null)
					rs.close();
			} catch (SQLException se) {
				se.printStackTrace();
			}
			try {
				if (stmt != null)
					stmt.close();
			} catch (SQLException se) {
				se.printStackTrace();
			}
			try {
				if (conn != null)
					conn.close();
			} catch (SQLException se) {
				se.printStackTrace();

			}
		}

	}

}
